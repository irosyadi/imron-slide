In mathematics, the Laplace transform, named after its inventor Pierre-Simon Laplace, is an integral transform that converts a function of a real variable \(t\) (often time) to a function of a complex variable \(s\) (complex frequency). The transform has many applications in science and engineering because it is a tool for solving differential equations. In particular, it transforms differential equations into algebraic equations and convolution into multiplication.

In this example, a simple ‘interactive table’ for Laplace transform is provided: upon function selection (via dropdown menu), the plot is generated for \(f(t)\), and the corresponding Laplace transform \(F(s)\) is given.

def first():
    with output_plot:
        linear_fun()
        
    with output_panel:

def linear_fun():   
    output_plot.clear_output()
    
    plt.figure(figsize=(10,5))
    x_axis = np.linspace(0,5, num=1000)
    plt.plot(x_axis, x_axis, 'b-')
    plt.xlim(0, 5)
    plt.ylim(0, 5)
    plt.grid(True)
    plt.legend(handles=[blue_patch])
    plt.xlabel('t [s]')
    plt.ylabel('f(t)')
    plt.xticks(range(0,5,1))
    plt.title(r'Graph of the function $f(t) = t$')
    plt.show()
    
    with output_panel:
        output_panel.clear_output()
        display(Markdown(r'Input function: $f(t) = t$'))
        display(Markdown(r'Laplace transform:'))
        display(Markdown(r'$\mathcal{L}\{F(t)\} = F(s) = \frac{1}{s^2}$'))
        display(Markdown(r'with $Re(s) > 0$'))

    
def constant_fun(c=1):
    output_plot.clear_output()
    display(c_slider)
    
    plt.figure(figsize=(10,5))
    x_axis = np.linspace(0,5, num=1000)
    plt.plot(x_axis, [c for t in x_axis], 'b-')
    plt.xlim(0, 5)
    plt.ylim(-1, 5)
    plt.grid(True)
    plt.legend(handles=[blue_patch])
    plt.xlabel('t [s]')
    plt.ylabel('f(t)')
    plt.xticks(range(0,5,1))
    plt.title(r'Graph of the function $f(t) = c$')
    plt.show()
        
    with output_panel:
        output_panel.clear_output(wait=True)
        display(Markdown(r'Input function: $f(t) = c$'))
        display(Markdown(r'Laplace transform:'))
        display(Markdown(r'$\mathcal{L}\{F(t)\} = F(s) = \frac{c}{s}$'))
        display(Markdown(r'with $Re(s) > 0$'))
        display(Markdown(r'For c = '+str(c_slider.value)+r': $F(s) = \frac{'+str(c_slider.value)+r'}{s}$'))
  
    
def potential_fun(n=1):
    output_plot.clear_output()
    display(n_slider)
    
    plt.figure(figsize=(10,5))
    x_axis = np.linspace(0,5, num=1000)
    plt.plot(x_axis, [x**n for x in x_axis], 'b-')
    plt.xlim(0, 5)
    plt.ylim(0, 20)
    plt.grid(True)
    plt.legend(handles=[blue_patch])
    plt.xlabel('t [s]')
    plt.ylabel('f(t)')
    plt.xticks(range(0,5,1))
    plt.title(r'Graph of the function $f(t) = t^n$')
    plt.show()
        
    with output_panel:
        output_panel.clear_output()
        display(Markdown(r'Input function: $f(t) = t^n$'))
        display(Markdown(r'Laplace transform:'))
        display(Markdown(r'$\mathcal{L}\{F(t)\} = F(s) = \frac{n!}{s^{n+1}}$'))
        display(Markdown(r'with $Re(s) > 0$'))
        display(Markdown(r'For n = '+str(n_slider.value)+r': $F(s)= \frac{'+str(math.factorial(n_slider.value))+r'}{s^'+str(n_slider.value+1)+r'}$'))

        
def exponential_fun(k):
    output_plot.clear_output()
    display(k_slider_exp)
    
    plt.figure(figsize=(10,5))
    x_axis = np.linspace(0,5, num=1000)
    plt.plot(x_axis, [np.e**(k*x) for x in x_axis], 'b-')
    plt.xlim(0, 4)
    plt.ylim(0, 20)
    plt.grid(True)
    plt.legend(handles=[blue_patch])
    plt.xlabel('t [s]')
    plt.ylabel('f(t)')
    plt.xticks(range(0,5,1))
    plt.title(r'Graph of the function $f(t) = e^{kt}$')
    plt.show()
        
    with output_panel:
        output_panel.clear_output()
        display(Markdown(r'Input function: $f(t) = e^{kt}$'))
        display(Markdown(r'Laplace transform:'))
        display(Markdown(r'$\mathcal{L}\{F(t)\} = F(s) = \frac{1}{s-k}$'))
        display(Markdown(r'with $Re(s) > Re(k)$'))
        display(Markdown(r'For k = '+str(k_slider_exp.value)+r': $F(s)= \frac{1}{s-'+str(k_slider_exp.value)+r'}$'))

        
def sin_fun(a):
    output_plot.clear_output()
    display(a_slider_sin)
    
    plt.figure(figsize=(10,5))
    x_axis = np.linspace(0,5, num=1000)
    plt.plot(x_axis, [np.sin(2*np.pi*a*x)for x in x_axis], 'b-')
    plt.xlim(0, 5)
    plt.ylim(-2, 2)
    plt.grid(True)
    plt.legend(handles=[blue_patch])
    plt.xlabel('t [s]')
    plt.ylabel('f(t)')
    plt.xticks(range(0,5,1))
    plt.title(r'Graph of the function $f(t) = sin(at)$')
    plt.show()

    with output_panel:
        output_panel.clear_output()
        display(Markdown(r'Input function: $f(t) = sin(at)$'))
        display(Markdown(r'Laplace transform:'))
        display(Markdown(r'$\mathcal{L}\{F(t)\} = F(s) = \frac{a}{s^2 + a^2}$'))
        display(Markdown(r'with $Re(s) > 0$'))
        display(Markdown(r'For a = '+str(a_slider_sin.value)+r': $F(s)= \frac{'+str(a_slider_sin.value)+r'}{s^2+'+str(a_slider_sin.value**2)+r'}$'))

def exp_sin_fun(a, k):
    output_plot.clear_output()
    display(a_slider_expsin, k_slider_expsin)
    
    plt.figure(figsize=(10,5))
    x_axis = np.linspace(0,5, num=1000)
    plt.plot(x_axis, [np.e**(-k*x)* np.sin(2*np.pi*a*x) for x in x_axis], 'b-')
    plt.xlim(0, 5)
    plt.ylim(-2, 2)
    plt.grid(True)
    plt.legend(handles=[blue_patch])
    plt.xlabel('t [s]')
    plt.ylabel('f(t)')
    plt.xticks(range(0,5,1))
    plt.title(r'Graph of the function $f(t) = e^{-kt} sin(at)$')
    plt.show()
    
    with output_panel:
        output_panel.clear_output()
        display(Markdown(r'Input function: $f(t) = e^{-kt} sin(at)$'))
        display(Markdown(r'Laplace transform:'))
        display(Markdown(r'$\mathcal{L}\{F(t)\} = F(s) = \frac{a}{(s+k)^2 + a^2}$'))
        display(Markdown(r'with $Re(s) > -k$'))
        display(Markdown(r'For a, k = '+str(a_slider_expsin.value)+', '+str(k_slider_expsin.value)+r': $F(s)= \frac{'+str(a_slider_expsin.value)+r'}{(s+'+str(k_slider_expsin.value)+r')^2+'+str(a_slider_expsin.value**2)+r'}$'))

# slider observers        
def slider_change(change, sender):
    if sender == 'constant':
        constant_fun(change['new'])
    if sender == 'potential':
        potential_fun(change['new'])
    if sender == 'exponential':
        exponential_fun(change['new'])
    if sender == 'sin':
        sin_fun(change['new'])
    if sender == 'cos':
        cos_fun(change['new'])
    if sender == 'expsin_a':
        exp_sin_fun(change['new'], k_slider_expsin.value)
    if sender == 'expsin_k':
        exp_sin_fun(a_slider_expsin.value, change['new'])
    if sender == 'expcos_a':
        exp_cos_fun(change['new'], k_slider_expcos.value)
    if sender == 'expcos_k':
        exp_cos_fun(a_slider_expcos.value, change['new'])

c_slider.observe(lambda change: slider_change(change, 'constant'), names='value')
n_slider.observe(lambda change: slider_change(change, 'potential'), names='value')
k_slider_exp.observe(lambda change: slider_change(change, 'exponential'), names='value')
a_slider_sin.observe(lambda change: slider_change(change, 'sin'), names='value')
a_slider_cos.observe(lambda change: slider_change(change, 'cos'), names='value')
a_slider_expsin.observe(lambda change: slider_change(change, 'expsin_a'), names='value')
k_slider_expsin.observe(lambda change: slider_change(change, 'expsin_k'), names='value')
a_slider_expcos.observe(lambda change: slider_change(change, 'expcos_a'), names='value')
k_slider_expcos.observe(lambda change: slider_change(change, 'expcos_k'), names='value')

    
# dropdown selection handler    
def example_select_handler(change):
    if example_select.value == 'linear function':
        with output_plot:
            linear_fun()
    
    if example_select.value == 'constant function':
        with output_plot:
             constant_fun(c_slider.value)
    
    if example_select.value == 'potential function':
        with output_plot:
            potential_fun(n_slider.value)        
                 
    if example_select.value == 'exponential function':
        with output_plot:
            exponential_fun(k_slider_exp.value)
     
    if example_select.value == 'sine function':
        with output_plot:
            sin_fun(a_slider_sin.value)
           
    if example_select.value =='cosine function':
        with output_plot:
            cos_fun(a_slider_cos.value)
    
    if example_select.value == 'damped sine function':
        with output_plot:
            exp_sin_fun(a_slider_expsin.value, k_slider_expsin.value)
        
    if example_select.value == 'damped cosine function':
        with output_plot:
            exp_cos_fun(a_slider_expcos.value, k_slider_expcos.value)
            
            
example_select.observe(example_select_handler, names='value')
display(example_select, widgets.HBox([output_plot, widgets.Label(" "), output_panel ]))
